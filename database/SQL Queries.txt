

Query 1: View All Items Inside a Specific Order


SELECT 
    o.order_id,
    o.status,
    c.item_id,
    c.item_name,
    c.category,
    c.price
FROM ORDERS o
JOIN CONTAINS ct ON o.order_id = ct.order_id
JOIN CLOTHING_ITEM c ON ct.item_id = c.item_id
WHERE o.order_id = 1;


Query 2: View All Items Sold by a Specific Designer


SELECT 
    d.designer_id,
    d.brand_name,
    c.item_name,
    c.price,
    o.order_id
FROM DESIGNER d
JOIN CLOTHING_ITEM c ON d.designer_id = c.designer_id
JOIN CONTAINS ct ON c.item_id = ct.item_id
JOIN ORDERS o ON ct.order_id = o.order_id
WHERE d.designer_id = 1;


Query 3: Count Number of Orders Per User


SELECT 
    u.user_id,
    u.user_name,
    COUNT(o.order_id) AS total_orders
FROM USER u
LEFT JOIN ORDERS o ON u.user_id = o.user_id
GROUP BY u.user_id, u.user_name;
